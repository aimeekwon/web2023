07. 데이터 이펙트 : 중첩 for문을 이용해서 구구단 만들기

                      
    for(let i=1; i<10; i++){
        document.write(i+"단","<br>")
        for(let j=1; j<10; j++){
            document.write(i+"*"+j+"="+i*j,"<br>")
        }
    }
